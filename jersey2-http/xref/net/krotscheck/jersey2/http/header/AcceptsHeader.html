18  package net.krotscheck.jersey2.http.header;
19  
20  import org.apache.http.HttpHeaders;
21  import org.apache.http.message.BasicHeader;
22  import org.glassfish.hk2.utilities.binding.AbstractBinder;
23  import org.glassfish.jersey.process.internal.RequestScoped;
24  
25  import javax.inject.Inject;
26  import javax.ws.rs.DefaultValue;
27  import javax.ws.rs.HeaderParam;
28  
29  /**
30   * The factory converts the raw HTTP Accept header into a BasicHeader collection
31   * of individual header elements, which are sorted based on priority. It is
32   * injected into the context as a Named entity using the header as its name.
33   *
34   * @author Michael Krotscheck
35   */
36  public final class AcceptsHeader extends BasicHeader {
37  
38      /**
39       * Construct a new instance of the header parameters.
40       *
41       * @param value The injected content of the Accept: header.
42       */
43      @Inject
44      protected AcceptsHeader(@HeaderParam(HttpHeaders.ACCEPT)
45                              @DefaultValue("")
46                              final String value) {
47          super(HttpHeaders.ACCEPT, value);
48      }
49  
50      /**
51       * HK2 Binder for our injector context.
52       */
53      public static final class Binder extends AbstractBinder {
54  
55          @Override
56          protected void configure() {
57              bind(AcceptsHeader.class)
58                      .to(AcceptsHeader.class)
59                      .to(BasicHeader.class)
60                      .named(HttpHeaders.ACCEPT)
61                      .in(RequestScoped.class);
62          }
63      }
64  }
