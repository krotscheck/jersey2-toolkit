18  package net.krotscheck.jersey2.hibernate.factory;
19  
20  import org.glassfish.hk2.api.Factory;
21  import org.glassfish.hk2.utilities.binding.AbstractBinder;
22  import org.glassfish.jersey.process.internal.RequestScoped;
23  import org.hibernate.search.FullTextSession;
24  import org.hibernate.search.SearchFactory;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  import javax.inject.Inject;
29  
30  /**
31   * This factory will generate a Hibernate Search SearchFactory from the current
32   * hibernate session. It is request-scoped, since it depends on a valid
33   * session.
34   *
35   * @author Michael Krotscheck
36   */
37  public final class FulltextSearchFactoryFactory
38          implements Factory<SearchFactory> {
39  
40      /**
41       * Logger instance.
42       */
43      private static Logger logger =
44              LoggerFactory.getLogger(FulltextSearchFactoryFactory.class);
45  
46      /**
47       * The current hibernate session injected from the application container.
48       */
49      private FullTextSession session;
50  
51      /**
52       * Create a new fulltext search factory.
53       *
54       * @param fullTextSession The fulltext session to use.
55       */
56      @Inject
57      public FulltextSearchFactoryFactory(final FullTextSession fullTextSession) {
58          this.session = fullTextSession;
59      }
60  
61      /**
62       * Create a request-scoped SearchFactory.
63       *
64       * @return A new SearchFactory
65       */
66      @Override
67      public SearchFactory provide() {
68          logger.trace("Creating hibernate search factory.");
69          return session.getSearchFactory();
70      }
71  
72      /**
73       * Dispose of the search factory.
74       *
75       * @param factory The factory to dispose of.
76       */
77      @Override
78      public void dispose(final SearchFactory factory) {
79          // Do nothing- the factory has no disposal requirements.
80      }
81  
82      /**
83       * HK2 Binder for our injector context.
84       */
85      public static final class Binder extends AbstractBinder {
86  
87          @Override
88          protected void configure() {
89              bindFactory(FulltextSearchFactoryFactory.class)
90                      .to(SearchFactory.class)
91                      .in(RequestScoped.class);
92          }
93      }
94  }
