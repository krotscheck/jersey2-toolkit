43  /**
44   * This factory provides a singleton Hibernate SessionFactory for your
45   * application context. You may use this to generate your own sessions if you
46   * are not in an explicit request context, however if you do so please make sure
47   * to clean up after yourself and call session.close().
48   * <p/>
49   * Furthermore, this injector will search through the ServiceLocator context to
50   * determine if there are any Hibernate Event Listeners injected from other
51   * features. It will gather those and automatically register them against the
52   * hibernate factory.
53   *
54   * @author Michael Krotscheck
55   */
56  public final class HibernateSessionFactoryFactory
57          implements Factory<SessionFactory> {
58  
59      /**
60       * Logger instance.
61       */
62      private static Logger logger = LoggerFactory
63              .getLogger(HibernateSessionFactoryFactory.class);
64  
65      /**
66       * Create a new factory factory.
67       *
68       * @param hibernateConfig The Hibernate Config
69       * @param serviceLocator  The service locator from which to resolve event
70       *                        handlers.
71       */
72      @Inject
73      public HibernateSessionFactoryFactory(final Configuration hibernateConfig,
74                                            final ServiceLocator serviceLocator) {
75          this.configuration = hibernateConfig;
76          this.locator = serviceLocator;
77      }
78  
79      /**
80       * Injected hibernate configuration.
81       */
82      private Configuration configuration;
83  
84      /**
85       * The service locator.
86       */
87      private ServiceLocator locator;
88  
89      /**
90       * Provide a singleton instance of the hibernate session factory.
91       *
92       * @return A session factory.
93       */
94      @Override
95      public SessionFactory provide() {
96          logger.trace("Creating hibernate session factory.");
97  
98          // Build the service registry.
99          ServiceRegistry registry = new StandardServiceRegistryBuilder()
100                 .applySettings(configuration.getProperties())
101                 .build();
102 
103         SessionFactory factory = configuration.buildSessionFactory(registry);
104 
105         // Register our event listeners.
106         injectEventListeners(((SessionFactoryImpl) factory)
107                 .getServiceRegistry());
108 
109         return factory;
110     }
111 
112     /**
113      * Dispose of the hibernate session.
114      *
115      * @param sessionFactory The session to dispose.
116      */
117     @Override
118     public void dispose(final SessionFactory sessionFactory) {
119         if (sessionFactory != null && !sessionFactory.isClosed()) {
120             logger.info("Disposing of hibernate session factory.");
121             sessionFactory.close();
122         }
123     }
124 
125     /**
126      * This method automatically adds discovered hibernate event listeners into
127      * the hibernate service registry.
128      *
129      * @param registry The service registry.
130      */
131     private void injectEventListeners(final ServiceRegistry registry) {
132 
133         EventListenerRegistry eventRegistry = registry
134                 .getService(EventListenerRegistry.class);
135 
136         List<PostInsertEventListener> postInsertEvents = locator
137                 .getAllServices(PostInsertEventListener.class);
138         if (postInsertEvents != null) {
139             for (PostInsertEventListener piEventListener : postInsertEvents) {
140                 logger.trace("Registering PostInsert: " + piEventListener
141                         .getClass().getCanonicalName());
142                 eventRegistry.appendListeners(EventType.POST_INSERT,
143                         piEventListener);
144             }
145         }
146 
147         List<PostUpdateEventListener> postUpdateEvents = locator
148                 .getAllServices(PostUpdateEventListener.class);
149         if (postUpdateEvents != null) {
150             for (PostUpdateEventListener puEventListener : postUpdateEvents) {
151                 logger.trace("Registering PostUpdate: " + puEventListener
152                         .getClass().getCanonicalName());
153                 eventRegistry.appendListeners(EventType.POST_UPDATE,
154                         puEventListener);
155             }
156         }
157 
158         List<PostDeleteEventListener> postDeleteEvents = locator
159                 .getAllServices(PostDeleteEventListener.class);
160         if (postDeleteEvents != null) {
161             for (PostDeleteEventListener pdEventListener : postDeleteEvents) {
162                 logger.trace("Registering PostDelete: " + pdEventListener
163                         .getClass().getCanonicalName());
164                 eventRegistry.appendListeners(EventType.POST_DELETE,
165                         pdEventListener);
166             }
167         }
168 
169         List<PreInsertEventListener> preInsertEvents = locator
170                 .getAllServices(PreInsertEventListener.class);
171         if (preInsertEvents != null) {
172             for (PreInsertEventListener piEventListener : preInsertEvents) {
173                 logger.trace("Registering PreInsert: " + piEventListener
174                         .getClass().getCanonicalName());
175                 eventRegistry.appendListeners(EventType.PRE_INSERT,
176                         piEventListener);
177             }
178         }
179 
180         List<PreUpdateEventListener> preUpdateEvents = locator
181                 .getAllServices(PreUpdateEventListener.class);
182         if (preUpdateEvents != null) {
183             for (PreUpdateEventListener puEventListener : preUpdateEvents) {
184                 logger.trace("Registering PreUpdate: " + puEventListener
185                         .getClass().getCanonicalName());
186                 eventRegistry.appendListeners(EventType.PRE_UPDATE,
187                         puEventListener);
188             }
189         }
190 
191         List<PreDeleteEventListener> preDeleteEvents = locator
192                 .getAllServices(PreDeleteEventListener.class);
193         if (preDeleteEvents != null) {
194             for (PreDeleteEventListener pdEventListener : preDeleteEvents) {
195                 logger.trace("Registering PreDelete: " + pdEventListener
196                         .getClass().getCanonicalName());
197                 eventRegistry.appendListeners(EventType.PRE_DELETE,
198                         pdEventListener);
199             }
200         }
201     }

203     /**
204      * HK2 Binder for our injector context.
205      */
206     public static final class Binder extends AbstractBinder {
207 
208         @Override
209         protected void configure() {
210             bindFactory(HibernateSessionFactoryFactory.class)
211                     .to(SessionFactory.class)
212                     .in(PerThread.class);
213         }
214     }
