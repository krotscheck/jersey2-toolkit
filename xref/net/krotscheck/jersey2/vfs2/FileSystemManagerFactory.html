18  package net.krotscheck.jersey2.vfs2;
19  
20  import org.apache.commons.vfs2.FileSystemException;
21  import org.apache.commons.vfs2.FileSystemManager;
22  import org.apache.commons.vfs2.VFS;
23  import org.glassfish.hk2.api.Factory;
24  import org.glassfish.hk2.utilities.binding.AbstractBinder;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  import javax.inject.Singleton;
29  
30  /**
31   * This factory injects a singleton instance of the file system manager into the
32   * jersey context.
33   *
34   * @author Michael Krotscheck
35   */
36  public final class FileSystemManagerFactory
37          implements Factory<FileSystemManager> {
38  
39      /**
40       * Logger instance.
41       */
42      private static Logger logger =
43              LoggerFactory.getLogger(FileSystemManagerFactory.class);
44  
45      /**
46       * Provide the file system manager.
47       *
48       * @return A file system manager instance.
49       */
50      @Override
51      public FileSystemManager provide() {
52          try {
53              return VFS.getManager();
54          } catch (FileSystemException fse) {
55              logger.error("Cannot create FileSystemManager", fse);
56              throw new RuntimeException("Cannot create FileSystemManager", fse);
57          }
58      }
59  
60      /**
61       * Dispose of the filesystem.
62       *
63       * @param manager The manager to dispose of.
64       */
65      @Override
66      public void dispose(final FileSystemManager manager) {
67          // Do nothing.
68      }
69  
70      /**
71       * HK2 Binder for our injector context.
72       */
73      public static final class Binder extends AbstractBinder {
74  
75          @Override
76          protected void configure() {
77              bindFactory(FileSystemManagerFactory.class)
78                      .to(FileSystemManager.class)
79                      .in(Singleton.class);
80          }
81      }
82  }
